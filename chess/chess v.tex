			\foreach \y in {1, 2, 3, 4, 5, 6, 7, 8}{
				\pgfmathtruncatemacro\xplusymodtwo{mod(\x + \y, 2)}
				\ifnum\xplusymodtwo = 1
					\fill [black!66!yellow] (\x, \y) rectangle (\x+1, \y+1);
				\fi
			}
		}
	\end{tikzpicture}
