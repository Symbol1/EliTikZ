\documentclass{article}

	\usepackage{xeCJK}
		\setCJKmainfont{NotoSansTC-Regular}
	\usepackage{tikz}
	\usepackage{listings}
\begin{document}



	\begin{tikzpicture}
		\draw[->] (-5, 0) -- (5, 0) node[right]{$x$ axis};
		\draw[->] (0, -5) -- (0, 5) node[above]{$y$ axis};
		\def\a{0.2}
		\def\b{0.5}
		\def\c{-2}
		\draw plot[variable = \t, domain = -5:5, samples = 100]
			({\t}, {\a*(\t)^2 + \b*\t + \c});
		\draw(0, \c) circle(1pt) node[below right]{$y$-intercept};
		\pgfmathsetmacro\rooti{
			(-\b + sqrt((\b)^2 - 4*\a*\c))
			/
			(2*\a)
		}
		\pgfmathsetmacro\rootii{
			(-\b - sqrt((\b)^2 - 4*\a*\c))
			/
			(2*\a)
		}
		\draw(\rooti, 0) circle(1pt) node[below right]{$x$-intercept};
		\draw(\rootii, 0) circle(1pt) node[above right]{$x$-another};
	\end{tikzpicture}
